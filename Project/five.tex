\begin{frame}
\begin{align*}
\large
P(car|switch) &= P \cdot \dfrac{n(n+m-1)}{(n+m)(n+m-k-1)} + (1-P) \cdot \dfrac{n}{n+m}\\ \\
&= \dfrac{n(n+m-k-1+Pk)}{(n+m)(n+m-k-1)}
\end{align*}
\end{frame}
