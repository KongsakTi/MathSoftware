\begin{theorem}
$$
\det(M) = \det
\left(
\begin{array}{cc}
M_1 & M_2 \\
M_3 & M_4 \\
\end{array}
\right)
= \det(M_1)\det(M_4 - M_3M_1^{-1}M_2).
$$
\end{theorem}

\begin{Proof}
The proof is not too hard. Try it yourself!
\end{Proof}
